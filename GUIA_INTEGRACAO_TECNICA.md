  async generateResponse(userMessage, userId = 'anonymous') {
    try {
      // Buscar hist√≥rico da conversa
      const history = this.conversationHistory.get(userId) || [];
      
      // Construir mensagens

        messages: messages,

      // Atualizar hist√≥rico

      // Manter apenas √∫ltimas 10 mensagens

        success: true,
        message: assistantMessage,
        usage: response.usage

        success: false,
        message: 'Desculpe, estou com dificuldades t√©cnicas. Tente novamente em instantes.',
